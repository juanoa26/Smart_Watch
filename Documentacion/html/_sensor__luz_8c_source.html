21uint16_t ReadLUZ(void) {
22
23 TRISD = 0x00;
24 TRISA = 0xFF;

26 OSCCONbits.IRCF = 0b111; //Configura oscilador interno (FOSC = 8Mhz)
27 OSCCONbits.SCS = 0b10; // Oscilador del sistema = Fosc interno

29 //ADC = 10bits, Tad = 1us, Tacq = 4us, Vref = 5v-0v, RA=ANALOG
30 ADCON1bits.PCFG = 0b1101; // Configura el Puerto como Entrada Analógica.
31 ADCON1bits.VCFG = 0b00; // Selecciona Voltajes de Referencia (5v-0v).
32 ADCON0bits.CHS = 0b0001; // Selecciona el Canal Analógico.
33 ADCON2bits.ACQT = 0b010; // Tiempo de Adquisición 4Tad.
34 ADCON2bits.ADCS = 0b001; // Tiempo de Conversión Fosc/8.
35 ADCON2bits.ADFM = 1; // Justificación derecha (modo-10bits).
36 ADCON0bits.ADON = 1; // Habilita el Módulo AD.
37
38 while (1) {
39 ADCON0bits.GO_DONE = 1; //Inicia la COnversió AD.
40 while (ADCON0bits.GO_DONE); // Espera a que termine la conversión AD.
41 result = ADRESH; // Lectura de valor AD.
42 result = (result << 8) + ADRESL;
